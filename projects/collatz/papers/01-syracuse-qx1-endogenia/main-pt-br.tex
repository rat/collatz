\[

$\xi\in\mathbb{Z}/3^n\mathbb{Z}$ com $3\nmid \xi$,
\[

\section{Preliminares: o mapa $qx+1$ acelerado e sua УЁrvore reversa}

\begin{example}[O autУДmato ingУЊnuo mod $q$ nУЃo estУЁ bem definido]
\label{ex:naive-fails}
Tome $q=3$ (logo $d=\OpOrd_3(2)=2$) e $a=2$, admissУ­vel para todo pai
$u\equiv1\pmod3$. Dois pais $u=1$ e $u=7$, ambos $\equiv1\pmod3$, dУЃo

conhecimento de $u$ mod $q^k$ fornece. NУЃo existe, portanto, um

O que \emph{estУЁ} bem definido, e УЉ o substituto correto do quadro
ingУЊnuo do Exemplo~\ref{ex:naive-fails}, УЉ uma bijeУЇУЃo entre uma fibra

\begin{lemma}[BijeУЇУЃo de fibra]\label{lem:fibre-bijection}
Fixe $a\ge1$ e $j\ge1$. O mapa $v\mapsto w_a(v):=(2^av-1)/q$ УЉ uma
bijeУЇУЃo de $\{v\bmod q^j : 2^av\equiv1\!\!\pmod q\}$ (uma progressУЃo
aritmУЉtica de $q^{j-1}$ resУ­duos mod $q^j$) sobre
$\mathbb{Z}/q^{j-1}\mathbb{Z}$.
\end{lemma}
\begin{proof}
Bem definido: $v\bmod q^j$ determina $2^av-1\bmod q^j$, logo
$w_a(v)\bmod q^{j-1}$. Injetivo: se $v\not\equiv v'\pmod{q^j}$ dentro
da progressУЃo, escreva $v-v'=qt$ com $t\not\equiv0\pmod{q^{j-1}}$;
entУЃo $w_a(v)-w_a(v')=2^at\not\equiv0\pmod{q^{j-1}}$ jУЁ que $2$ УЉ
unidade mod $q^{j-1}$. A sobrejetividade segue por ser uma injeУЇУЃo
entre dois conjuntos finitos de mesmo tamanho $q^{j-1}$.

Iterando o Lema~\ref{lem:fibre-bijection} ao longo de um caminho de
profundidade $k$ $(a_1,\dots,a_k)$ recupera-se, em forma truncada,
exatamente a identidade autossimilar \eqref{eq:X-selfsimilar} da
\S\ref{sec:context} abaixo: toda sequУЊncia
$(a_1,\dots,a_k)\in(\mathbb{Z}_{\ge1})^k$, sem restriУЇУЃo de
admissibilidade sobre os $a_i$ individuais, УЉ admissУ­vel para
\emph{exatamente uma} raiz-residual $u_0\bmod q^k$, a saber
$u_0\equiv\sum_{i=1}^k q^{i-1}2^{-S_i}\pmod{q^k}$,
$S_i:=a_1+\cdots+a_i$. Defina a funУЇУЃo de partiУЇУЃo em profundidade $k$
enraizada em $u_0$,
\[
  Z_k(\alpha;u_0) := \sum_{\substack{(a_1,\dots,a_k)\text{ admissУ­vel}\\ \text{a partir de } u_0}} \ \prod_{i=1}^k \bigl(q\cdot2^{-a_i}\bigr)^\alpha .

\begin{theorem}[Identidade de pressУЃo anelada exata]\label{thm:pressure}

\begin{equation}\label{eq:annealed-identity}

\end{equation}

\begin{equation}\label{eq:pressure-closed-form}

\end{equation}

\begin{equation}\label{eq:pressure-eq}

\end{equation}
\end{theorem}
\begin{proof}

\end{proof}
